175 Graphics::Color World::refractedColor(const ComputedValues& cv, int remainingCalls) const
176 {
177  if (remainingCalls == 0)
178  {
179  return Graphics::Color::black();
180  }
181 
182  if (cv.shape().material().transparency == 0)
183  {
184  return Graphics::Color::black();
185  }
186 
187  float nRatio = cv.n1() / cv.n2();
188 
189  float cosI = static_cast<float>(Math::Vector::dot(cv.eye(), cv.normal()));
190  float sin2t = powf(nRatio, 2.0f) * (1 - powf(cosI, 2.0f));
191 
192  if (sin2t > 1)
193  {
194  return Graphics::Color::black();
195  }
196 
197  float cosT = sqrtf(1.0f - sin2t);

199  Math::Vector direction = cv.normal() * (nRatio * cosI - cosT) - cv.eye() * nRatio;
200 
201  Math::Ray refractedRay = Math::Ray(cv.underPosition(), direction);

206 bool World::isShadowed(const Math::Point& position, const PointLight& light) const
207 {
208  Math::Vector positionToLight = light.position() - position;
209  double distance = positionToLight.magnitude();
210  Math::Vector positionToLightDirection = positionToLight.normalize();
211 
212  Math::Ray ray = Math::Ray(position, positionToLightDirection);
213  Math::Intersections intersections = intersectRay(ray);
214 
215  if (!intersections.hit().has_value())
216  {
217  return false;
218  }
219 
220  const Math::Intersection& hit = intersections.hit().value();
221 
222  if (hit.t() < distance)

224  const Renderer::Shape& shape = static_cast<const Renderer::Shape&>(hit.shape());
225  auto mode = shape.material().shadowcastMode;
226 
227  if (mode == ShadowcastMode::ALWAYS || (mode == ShadowcastMode::WHEN_TRANSPARENT && shape.material().transparency == 0))
228  {
229  return true;
230  }
231  else
